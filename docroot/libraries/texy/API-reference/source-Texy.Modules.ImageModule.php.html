106:     {
107:         list(, $mURLs, $mMod, $mAlign, $mLink) = $matches;

159:      * Parses image's syntax.
160:      * @param  string  input: small.jpg 80x13 | small-over.jpg | linked.jpg
161:      * @param  string
162:      * @param  bool
163:      * @return Image
164:      */
165:     public function factoryImage($content, $mod, $tryRef = TRUE)
166:     {
167:         $image = $tryRef ? $this->getReference(trim($content)) : FALSE;
168: 
169:         if (!$image) {
170:             $texy = $this->texy;
171:             $content = explode('|', $content);
172:             $image = new Image;
173: 
174:             // dimensions
175:             $matches = NULL;
176:             if ($matches = Texy\Regexp::match($content[0], '#^(.*) (\d+|\?) *(X|x) *(\d+|\?) *()$#U')) {
177:                 $image->URL = trim($matches[1]);
178:                 $image->asMax = $matches[3] === 'X';
179:                 $image->width = $matches[2] === '?' ? NULL : (int) $matches[2];
180:                 $image->height = $matches[4] === '?' ? NULL : (int) $matches[4];
181:             } else {
182:                 $image->URL = trim($content[0]);
183:             }
184: 
185:             if (!$texy->checkURL($image->URL, $texy::FILTER_IMAGE)) {
186:                 $image->URL = NULL;
187:             }
188: 
189:             // onmouseover image
190:             if (isset($content[1])) {
191:                 $tmp = trim($content[1]);
192:                 if ($tmp !== '' && $texy->checkURL($tmp, $texy::FILTER_IMAGE)) {
193:                     $image->overURL = $tmp;
194:                 }
195:             }
196: 
197:             // linked image
198:             if (isset($content[2])) {
199:                 $tmp = trim($content[2]);
200:                 if ($tmp !== '' && $texy->checkURL($tmp, $texy::FILTER_ANCHOR)) {
201:                     $image->linkedURL = $tmp;
202:                 }
203:             }
204:         }
205: 
206:         $image->modifier->setProperties($mod);

211:     /**
212:      * Finish invocation.
213:      * @return Texy\HtmlElement|FALSE
214:      */
215:     public function solve(Texy\HandlerInvocation $invocation = NULL, Image $image, Texy\Link $link = NULL)
216:     {
217:         if ($image->URL == NULL) {
218:             return FALSE;
219:         }
220: 
221:         $texy = $this->texy;
222: 
223:         $mod = $image->modifier;
224:         $alt = $mod->title;
225:         $mod->title = NULL;
226:         $hAlign = $mod->hAlign;
227:         $mod->hAlign = NULL;
228: 
229:         $el = new Texy\HtmlElement('img');
230:         $el->attrs['src'] = NULL; // trick - move to front
231:         $mod->decorate($texy, $el);
232:         $el->attrs['src'] = Helpers::prependRoot($image->URL, $this->root);
233:         if (!isset($el->attrs['alt'])) {
234:             $el->attrs['alt'] = $alt === NULL ? $this->defaultAlt : $texy->typographyModule->postLine($alt);
235:         }
236: 
237:         if ($hAlign) {
238:             $var = $hAlign . 'Class'; // leftClass, rightClass
239:             if (!empty($this->$var)) {
240:                 $el->attrs['class'][] = $this->$var;
241: 
242:             } elseif (empty($texy->alignClasses[$hAlign])) {
243:                 $el->attrs['style']['float'] = $hAlign;
244: 
245:             } else {
246:                 $el->attrs['class'][] = $texy->alignClasses[$hAlign];
247:             }
248:         }
249: 
250:         if (!is_int($image->width) || !is_int($image->height) || $image->asMax) {
251:             // autodetect fileRoot
252:             if ($this->fileRoot === NULL && isset($_SERVER['SCRIPT_FILENAME'])) {
253:                 $this->fileRoot = dirname($_SERVER['SCRIPT_FILENAME']) . '/' . $this->root;
254:             }
255: 
256:             // detect dimensions
257:             // absolute URL & security check for double dot
258:             if (Helpers::isRelative($image->URL) && strpos($image->URL, '..') === FALSE) {
259:                 $file = rtrim($this->fileRoot, '/\\') . '/' . $image->URL;
260:                 if (@is_file($file)) { // intentionally @
261:                     $size = @getImageSize($file); // intentionally @
262:                     if (is_array($size)) {
263:                         if ($image->asMax) {
264:                             $ratio = 1;
265:                             if (is_int($image->width)) {
266:                                 $ratio = min($ratio, $image->width / $size[0]);
267:                             }
268:                             if (is_int($image->height)) {
269:                                 $ratio = min($ratio, $image->height / $size[1]);
270:                             }
271:                             $image->width = round($ratio * $size[0]);
272:                             $image->height = round($ratio * $size[1]);
273: 
274:                         } elseif (is_int($image->width)) {
275:                             $image->height = round($size[1] / $size[0] * $image->width);
276: 
277:                         } elseif (is_int($image->height)) {
278:                             $image->width = round($size[0] / $size[1] * $image->height);
279: 
280:                         } else {
281:                             $image->width = $size[0];
282:                             $image->height = $size[1];
283:                         }
284:                     }
285:                 }
286:             }
287:         }
288: 
289:         $el->attrs['width'] = $image->width;
290:         $el->attrs['height'] = $image->height;
291: 
292:         // onmouseover actions generate
293:         if (!empty($texy->allowed['image/hover']) && $image->overURL !== NULL) {
294:             $overSrc = Helpers::prependRoot($image->overURL, $this->root);
295:             $el->attrs['onmouseover'] = 'this.src=\'' . addSlashes($overSrc) . '\'';
296:             $el->attrs['onmouseout'] = 'this.src=\'' . addSlashes($el->attrs['src']) . '\'';
297:             $el->attrs['onload'] = str_replace('%i', addSlashes($overSrc), $this->onLoad);
298:             $texy->summary['preload'][] = $overSrc;
299:         }
300: 
301:         $texy->summary['images'][] = $el->attrs['src'];
302: 
303:         if ($link) {
304:             return $texy->linkModule->solve(NULL, $link, $el);
305:         }
306: 
307:         return $el;
308:     }
