 42:     public function __construct($texy)
 43:     {
 44:         $this->texy = $texy;
 45:         $texy->allowed['emoticon'] = FALSE;
 46:         $texy->addHandler('emoticon', [$this, 'solve']);
 47:         $texy->addHandler('beforeParse', [$this, 'beforeParse']);
 48:     }

 51:     public function beforeParse()
 52:     {
 53:         if (empty($this->texy->allowed['emoticon'])) {
 54:             return;
 55:         }
 56: 
 57:         krsort($this->icons);
 58: 
 59:         $pattern = [];
 60:         foreach ($this->icons as $key => $foo) {
 61:             $pattern[] = preg_quote($key, '#') . '+'; // last char can be repeated
 62:         }
 63: 
 64:         $this->texy->registerLinePattern(
 65:             [$this, 'pattern'],
 66:             '#(?<=^|[\\x00-\\x20])(' . implode('|', $pattern) . ')#',
 67:             'emoticon',
 68:             '#' . implode('|', $pattern) . '#'
 69:         );
 70:     }

 73:     /**
 74:      * Callback for: :-))).
 75:      * @return Texy\HtmlElement|string|FALSE
 76:      */
 77:     public function pattern(Texy\LineParser $parser, array $matches)
 78:     {
 79:         $match = $matches[0];
 80: 
 81:         // find the closest match
 82:         foreach ($this->icons as $emoticon => $foo) {
 83:             if (strncmp($match, $emoticon, strlen($emoticon)) === 0) {
 84:                 return $this->texy->invokeAroundHandlers('emoticon', $parser, [$emoticon, $match]);
 85:             }
 86:         }
 87: 
 88:         return FALSE; // tohle se nestane
 89:     }

 92:     /**
 93:      * Finish invocation.

 95:      */
 96:     public function solve(Texy\HandlerInvocation $invocation, $emoticon, $raw)
 97:     {
 98:         $texy = $this->texy;
 99:         $file = $this->icons[$emoticon];
100:         $el = new Texy\HtmlElement('img');
101:         $el->attrs['src'] = Texy\Helpers::prependRoot($file, $this->root === NULL ? $texy->imageModule->root : $this->root);
102:         $el->attrs['alt'] = $raw;
103:         $el->attrs['class'][] = $this->class;
104: 
105:         // file path
106:         $file = rtrim($this->fileRoot === NULL ? $texy->imageModule->fileRoot : $this->fileRoot, '/\\') . '/' . $file;
107:         if (@is_file($file)) { // intentionally @
108:             $size = @getImageSize($file); // intentionally @
109:             if (is_array($size)) {
110:                 $el->attrs['width'] = $size[0];
111:                 $el->attrs['height'] = $size[1];
112:             }
113:         }
114:         $texy->summary['images'][] = $el->attrs['src'];
115:         return $el;
