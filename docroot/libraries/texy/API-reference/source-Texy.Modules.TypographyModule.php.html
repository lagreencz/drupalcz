 67:     public function beforeParse(Texy\Texy $texy, & $text)

 69:         // CONTENT_MARKUP mark: \x17-\x1F
 70:         // CONTENT_REPLACED mark: \x16
 71:         // CONTENT_TEXTUAL mark: \x17
 72:         // CONTENT_BLOCK: not used in postLine
 73: 
 74:         if (isset(self::$locales[$this->locale])) {
 75:             $locale = self::$locales[$this->locale];
 76:         } else { // fall back
 77:             $locale = self::$locales['en'];
 78:         }
 79: 
 80:         $this->pattern = [
 81:             '#(?<![.\x{2026}])\.{3,4}(?![.\x{2026}])#mu' => "\xe2\x80\xa6",                // ellipsis  ...
 82:             '#(?<=[\d ]|^)-(?=[\d ]|$)#'              => "\xe2\x80\x93",                   // en dash 123-123
 83:             '#(?<=[^!*+,/:;<=>@\\\\_|-])--(?=[^!*+,/:;<=>@\\\\_|-])#' => "\xe2\x80\x93",   // en dash alphanum--alphanum
 84:             '#,-#'                                    => ",\xe2\x80\x93",                  // en dash ,-
 85:             '#(?<!\d)(\d{1,2}\.) (\d{1,2}\.) (\d\d)#' => "\$1\xc2\xa0\$2\xc2\xa0\$3",      // date 23. 1. 1978

 87:             '# --- #'                                 => "\xc2\xa0\xe2\x80\x94 ",          // em dash ---
 88:             '# ([\x{2013}\x{2014}])#u'                => "\xc2\xa0\$1",                    // &nbsp; behind dash (dash stays at line end)

 90:             '#-{1,}> #'                               => "\xe2\x86\x92 ",                  // right arrow -->
 91:             '# <-{1,}#'                               => " \xe2\x86\x90 ",                 // left arrow <--
