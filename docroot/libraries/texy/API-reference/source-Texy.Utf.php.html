 80:     public static function utf2html($s, $encoding)
 81:     {
 82:         // convert from UTF-8
 83:         if (strcasecmp($encoding, 'utf-8') === 0) {
 84:             return $s;
 85:         }
 86: 
 87:         // prepare UTF-8 -> charset table
 88:         self::$xlat = & self::$xlatCache[strtolower($encoding)];
 89:         if (!self::$xlat) {
 90:             for ($i = 128; $i < 256; $i++) {
 91:                 $ch = @iconv($encoding, 'UTF-8//IGNORE', chr($i)); // intentionally @
 92:                 if ($ch) {
 93:                     self::$xlat[$ch] = chr($i);
 94:                 }
 95:             }
 96:         }
