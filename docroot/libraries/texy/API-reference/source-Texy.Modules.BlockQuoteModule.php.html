105:     /**
106:      * Converts cite source to URL.
107:      * @param  string
108:      * @return string|NULL
109:      */
110:     public function citeLink($link)
111:     {
112:         $texy = $this->texy;
113: 
114:         if ($link == NULL) {
115:             return NULL;
116:         }
117: 
118:         if ($link{0} === '[') { // [ref]
119:             $link = substr($link, 1, -1);
120:             $ref = $texy->linkModule->getReference($link);
121:             if ($ref) {
122:                 return Texy\Helpers::prependRoot($ref->URL, $texy->linkModule->root);
123:             }
124:         }
125: 
126:         // special supported case
127:         if (strncasecmp($link, 'www.', 4) === 0) {
128:             return 'http://' . $link;
129:         }
130: 
131:         return Texy\Helpers::prependRoot($link, $texy->linkModule->root);
132:     }
