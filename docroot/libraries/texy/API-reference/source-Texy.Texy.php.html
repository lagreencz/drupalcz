379:     /**
380:      * Converts document in Texy! to (X)HTML code.
381:      *
382:      * @param  string   input text
383:      * @param  bool     is single line?
384:      * @return string   output HTML code
385:      */
386:     public function process($text, $singleLine = FALSE)
387:     {
388:         if ($this->processing) {
389:             throw new \RuntimeException('Processing is in progress yet.');
390:         }
391: 
392:         // initialization
393:         $this->marks = [];
394:         $this->processing = TRUE;
395: 
396:         // speed-up
397:         if (is_array($this->allowedClasses)) {
398:             $this->_classes = array_flip($this->allowedClasses);
399:         } else {
400:             $this->_classes = $this->allowedClasses;
401:         }
402:         if (is_array($this->allowedStyles)) {
403:             $this->_styles = array_flip($this->allowedStyles);
404:         } else {
405:             $this->_styles = $this->allowedStyles;
406:         }
407: 
408:         // convert to UTF-8 (and check source encoding)
409:         $text = Utf::toUtf($text, $this->encoding);
410: 
411:         if ($this->removeSoftHyphens) {
412:             $text = str_replace("\xC2\xAD", '', $text);
413:         }
414: 
415:         // standardize line endings and spaces
416:         $text = Helpers::normalize($text);
417: 
418:         // replace tabs with spaces
419:         $this->tabWidth = max(1, (int) $this->tabWidth);
420:         while (strpos($text, "\t") !== FALSE) {
421:             $text = Regexp::replace($text, '#^([^\t\n]*+)\t#mU', function ($m) {
422:                 return $m[1] . str_repeat(' ', $this->tabWidth - strlen($m[1]) % $this->tabWidth);
423:             });
424:         }
425: 
426:         // user before handler
427:         $this->invokeHandlers('beforeParse', [$this, & $text, $singleLine]);
428: 
429:         // select patterns
430:         $this->_linePatterns = $this->linePatterns;
431:         $this->_blockPatterns = $this->blockPatterns;
432:         foreach ($this->_linePatterns as $name => $foo) {
433:             if (empty($this->allowed[$name])) {
434:                 unset($this->_linePatterns[$name]);
435:             }
436:         }
437:         foreach ($this->_blockPatterns as $name => $foo) {
438:             if (empty($this->allowed[$name])) {
439:                 unset($this->_blockPatterns[$name]);
440:             }
441:         }
442: 
443:         // parse Texy! document into internal DOM structure
444:         $this->DOM = new HtmlElement;
445:         if ($singleLine) {
446:             $this->DOM->parseLine($this, $text);
447:         } else {
448:             $this->DOM->parseBlock($this, $text);
449:         }
450: 
451:         // user after handler
452:         $this->invokeHandlers('afterParse', [$this, $this->DOM, $singleLine]);
453: 
454:         // converts internal DOM structure to final HTML code
455:         $html = $this->DOM->toHtml($this);
456: 
457:         // created by ParagraphModule and then protected
458:         $html = str_replace("\r", "\n", $html);
459: 
460:         $this->processing = FALSE;
461: 
462:         return Utf::utf2html($html, $this->encoding);
463:     }

516:     /**
517:      * Converts internal string representation to final HTML code in UTF-8.
518:      * @return string
519:      */
520:     final public function stringToHtml($s)
521:     {
522:         // decode HTML entities to UTF-8
523:         $s = html_entity_decode($s, ENT_QUOTES, 'UTF-8');
524: 
525:         // line-postprocessing
526:         $blocks = explode(self::CONTENT_BLOCK, $s);
527:         foreach ($this->postHandlers as $name => $handler) {
528:             if (empty($this->allowed[$name])) {
529:                 continue;
530:             }
531:             foreach ($blocks as $n => $s) {
532:                 if ($n % 2 === 0 && $s !== '') {
533:                     $blocks[$n] = call_user_func($handler, $s);
534:                 }
535:             }
536:         }
537:         $s = implode(self::CONTENT_BLOCK, $blocks);
538: 
539:         // encode < > &
540:         $s = htmlspecialchars($s, ENT_NOQUOTES, 'UTF-8');
541: 
542:         // replace protected marks
543:         $s = $this->unProtect($s);
544: 
545:         // wellform and reformat HTML
546:         $this->invokeHandlers('postProcess', [$this, & $s]);
547: 
548:         // unfreeze spaces
549:         $s = Helpers::unfreezeSpaces($s);
550:         $s = ltrim($s, "\n");
551: 
552:         return $s;
553:     }

556:     /**
557:      * Converts internal string representation to final HTML code in UTF-8.
558:      * @return string
559:      */
560:     final public function stringToText($s)
561:     {
562:         $save = $this->htmlOutputModule->lineWrap;
563:         $this->htmlOutputModule->lineWrap = FALSE;
564:         $s = $this->stringToHtml($s);
565:         $this->htmlOutputModule->lineWrap = $save;
566: 
567:         // remove tags
568:         $s = Regexp::replace($s, '#<(script|style)(.*)</\\1>#Uis', '');
569:         $s = strip_tags($s);
570:         $s = Regexp::replace($s, '#\n\s*\n\s*\n[\n\s]*\n#', "\n\n");
571: 
572:         // entities -> chars
573:         $s = html_entity_decode($s, ENT_QUOTES, 'UTF-8');
574: 
575:         // convert nbsp to normal space and remove shy
576:         $s = strtr($s, [
577:             "\xC2\xAD" => '', // shy
578:             "\xC2\xA0" => ' ', // nbsp
579:         ]);
580: 
581:         return $s;
582:     }

641:     {
