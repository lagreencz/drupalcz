 23:         AFTER = 2; // hyphenate after

 27:     private $consonants = [

 31:         "\xc4\x8c","\xc4\x8e","\xc5\x87","\xc5\x98","\xc5\xa0","\xc5\xa4","\xc5\xbd"];

 34:         'a','e','i','o','u','y',
 35:         'A','E','I','O','U','Y',

 77:     /**
 78:      * Callback for long words.
 79:      * @return string
 80:      * @internal
 81:      */
 82:     public function pattern(array $matches)
 83:     {
 84:         list($mWord) = $matches;
 85:         // [0] => lllloooonnnnggggwwwoorrdddd
 86: 
 87:         if (iconv_strlen($mWord, 'UTF-8') > self::SAFE_LIMIT) {
 88:             return $mWord;
 89:         }
 90: 
 91:         $chars = [];
 92:         preg_match_all(
 93:             '#['.Texy\Patterns::MARK.']+|.#u',
 94:             $mWord,
 95:             $chars
 96:         );
 97: 
 98:         $chars = $chars[0];
 99:         if (count($chars) < $this->wordLimit) {
100:             return $mWord;
101:         }
102: 
103:         $consonants = $this->consonants;
104:         $vowels = $this->vowels;
105:         $before_r = $this->before_r;
106:         $before_l = $this->before_l;
107:         $before_h = $this->before_h;
108:         $doubleVowels = $this->doubleVowels;
109: 
110:         $s = [];
111:         $trans = [];
112: 
113:         $s[] = '';
114:         $trans[] = -1;
115:         foreach ($chars as $key => $char) {
116:             if (ord($char{0}) < 32) {
117:                 continue;
118:             }
119:             $s[] = $char;
120:             $trans[] = $key;
121:         }
122:         $s[] = '';
123:         $len = count($s) - 2;
124: 
125:         $positions = [];
126:         $a = 0;
127:         $last = 1;
128: 
129:         while (++$a < $len) {
130:             $hyphen = self::DONT; // Do not hyphenate
131:             do {
132:                 if ($s[$a] === "\xC2\xA0") {
133:                     $a++;
134:                     continue 2;  // here and after never
135:                 }
136: 
137:                 if ($s[$a] === '.') {
138:                     $hyphen = self::HERE;
139:                     break;
140:                 }
141: 
142:                 if (isset($consonants[$s[$a]])) { // consonants
143: 
144:                     if (isset($vowels[$s[$a + 1]])) {
145:                         if (isset($vowels[$s[$a - 1]])) {
146:                             $hyphen = self::HERE;
147:                         }
148:                         break;
149:                     }
150: 
151:                     if (($s[$a] === 's') && ($s[$a - 1] === 'n') && isset($consonants[$s[$a + 1]])) {
152:                         $hyphen = self::AFTER;
153:                         break;
154:                     }
155: 
156:                     if (isset($consonants[$s[$a + 1]]) && isset($vowels[$s[$a - 1]])) {
157:                         if ($s[$a + 1] === 'r') {
158:                             $hyphen = isset($before_r[$s[$a]]) ? self::HERE : self::AFTER;
159:                             break;
160:                         }
161: 
162:                         if ($s[$a + 1] === 'l') {
163:                             $hyphen = isset($before_l[$s[$a]]) ? self::HERE : self::AFTER;
164:                             break;
165:                         }
166: 
167:                         if ($s[$a + 1] === 'h') { // CH
168:                             $hyphen = isset($before_h[$s[$a]]) ? self::DONT : self::AFTER;
169:                             break;
170:                         }
171: 
172:                         $hyphen = self::AFTER;
173:                         break;
174:                     }
175: 
176:                     break;
177:                 } // end of consonants
178: 
179:                 if (($s[$a] === 'u') && isset($doubleVowels[$s[$a - 1]])) {
180:                     $hyphen = self::AFTER;
181:                     break;
182:                 }
183:                 if (isset($vowels[$s[$a]]) && isset($vowels[$s[$a - 1]])) {
184:                     $hyphen = self::HERE;
185:                     break;
186:                 }
187:             } while (0);
188: 
189:             if ($hyphen === self::DONT && ($a - $last > $this->wordLimit * 0.6)) {
190:                 $positions[] = $last = $a - 1; // Hyphenate here
191:             }
192:             if ($hyphen === self::HERE) {
193:                 $positions[] = $last = $a - 1; // Hyphenate here
194:             }
195:             if ($hyphen === self::AFTER) {
196:                 $positions[] = $last = $a;
197:                 $a++; // Hyphenate after
198:             }
199:         } // while
200: 
201: 
202:         $a = end($positions);
203:         if (($a === $len - 1) && isset($consonants[$s[$len]])) {
204:             array_pop($positions);
205:         }
206: 
207:         $syllables = [];
208:         $last = 0;
209:         foreach ($positions as $pos) {
210:             if ($pos - $last > $this->wordLimit * 0.6) {
211:                 $syllables[] = implode('', array_splice($chars, 0, $trans[$pos] - $trans[$last]));
212:                 $last = $pos;
213:             }
214:         }
215:         $syllables[] = implode('', $chars);
216: 
217:         //$s = implode("\xC2\xAD", $syllables); // insert shy
218:         //$s = str_replace(array("\xC2\xAD\xC2\xA0", "\xC2\xA0\xC2\xAD"), array(' ', ' '), $s); // shy+nbsp = normal space
219: 
220:         return implode("\xC2\xAD", $syllables);
221:     }
